001package org.xbib.elasticsearch.common.termlist;
002
003import java.util.AbstractSet;
004import java.util.ConcurrentModificationException;
005import java.util.Iterator;
006import java.util.NoSuchElementException;
007import java.util.Set;
008
009/**
010 * CompactHashSet - implementation taken from
011 * http://code.google.com/p/ontopia/source/browse/trunk/ontopia/src/java/net/ontopia/utils/CompactHashSet.java?r=1704
012 * (License Apache 2.0)
013 * load factor increased to 0.9
014 */
015public class CompactHashSet<E> extends AbstractSet<E>
016        implements Set<E> {
017
018    private final static int INITIAL_SIZE = 3;
019
020    private final static double LOAD_FACTOR = 0.9;
021
022    private final E nullObject = (E) new Object();
023
024    private final E deletedObject = (E) new Object();
025
026    private int elements;
027
028    private int freecells;
029
030    private E[] objects;
031
032    private int modCount;
033
034    public CompactHashSet() {
035        objects = (E[]) new Object[INITIAL_SIZE];
036        elements = 0;
037        freecells = objects.length;
038        modCount = 0;
039    }
040
041    public CompactHashSet(int size) {
042        objects = (E[]) new Object[(size == 0 ? 1 : size)];
043        elements = 0;
044        freecells = objects.length;
045        modCount = 0;
046    }
047
048    public Iterator<E> iterator() {
049        return new CompactHashIterator();
050    }
051
052    public int size() {
053        return elements;
054    }
055
056    @Override
057    public boolean isEmpty() {
058        return elements == 0;
059    }
060
061    @Override
062    public boolean contains(Object o) {
063        if (o == null) {
064            o = nullObject;
065        }
066        int hash = o.hashCode();
067        int index = (hash & 0x7FFFFFFF) % objects.length;
068        int offset = 1;
069        while (objects[index] != null
070                && !(objects[index].hashCode() == hash
071                && objects[index].equals(o))) {
072            index = ((index + offset) & 0x7FFFFFFF) % objects.length;
073            offset = offset * 2 + 1;
074
075            if (offset == -1) {
076                offset = 2;
077            }
078        }
079        return objects[index] != null;
080    }
081
082    @Override
083    public boolean add(E e) {
084        if (e == null) {
085            e = nullObject;
086        }
087        int hash = e.hashCode();
088        int index = (hash & 0x7FFFFFFF) % objects.length;
089        int offset = 1;
090        int deletedix = -1;
091        while (objects[index] != null
092                && !(objects[index].hashCode() == hash
093                && objects[index].equals(e))) {
094            if (objects[index] == deletedObject) {
095                deletedix = index;
096            }
097            index = ((index + offset) & 0x7FFFFFFF) % objects.length;
098            offset = offset * 2 + 1;
099            if (offset == -1) {
100                offset = 2;
101            }
102        }
103        if (objects[index] == null) {
104            if (deletedix != -1) {
105                index = deletedix;
106            } else {
107                freecells--;
108            }
109            modCount++;
110            elements++;
111            objects[index] = e;
112            if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
113                rehash(objects.length);
114                if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
115                    rehash(objects.length * 2 + 1);
116                }
117            }
118            return true;
119        } else {
120            return false;
121        }
122    }
123
124    @Override
125    public boolean remove(Object o) {
126        if (o == null) {
127            o = nullObject;
128        }
129        int hash = o.hashCode();
130        int index = (hash & 0x7FFFFFFF) % objects.length;
131        int offset = 1;
132        while (objects[index] != null
133                && !(objects[index].hashCode() == hash
134                && objects[index].equals(o))) {
135            index = ((index + offset) & 0x7FFFFFFF) % objects.length;
136            offset = offset * 2 + 1;
137            if (offset == -1) {
138                offset = 2;
139            }
140        }
141        if (objects[index] != null) {
142            objects[index] = deletedObject;
143            modCount++;
144            elements--;
145            return true;
146        } else {
147            return false;
148        }
149    }
150
151    @Override
152    public void clear() {
153        elements = 0;
154        for (int ix = 0; ix < objects.length; ix++) {
155            objects[ix] = null;
156        }
157        freecells = objects.length;
158        modCount++;
159    }
160
161    @Override
162    public Object[] toArray() {
163        Object[] result = new Object[elements];
164        Object[] obj = this.objects;
165        int pos = 0;
166        for (int i = 0; i < obj.length; i++) {
167            if (obj[i] != null && obj[i] != deletedObject) {
168                if (obj[i] == nullObject) {
169                    result[pos++] = null;
170                } else {
171                    result[pos++] = obj[i];
172                }
173            }
174        }
175        return result;
176    }
177
178    @Override
179    public Object[] toArray(Object a[]) {
180        int size = elements;
181        if (a.length < size) {
182            a = (Object[]) java.lang.reflect.Array.newInstance(
183                    a.getClass().getComponentType(), size);
184        }
185        Object[] obj = this.objects;
186        int pos = 0;
187        for (int i = 0; i < obj.length; i++) {
188            if (obj[i] != null && obj[i] != deletedObject) {
189                if (obj[i] == nullObject) {
190                    a[pos++] = null;
191                } else {
192                    a[pos++] = obj[i];
193                }
194            }
195        }
196        return a;
197    }
198
199    protected void rehash(int newCapacity) {
200        Object[] newObjects = new Object[newCapacity];
201        for (Object o : objects) {
202            if (o == null || o == deletedObject) {
203                continue;
204            }
205            int hash = o.hashCode();
206            int index = (hash & 0x7FFFFFFF) % newCapacity;
207            int offset = 1;
208            while (newObjects[index] != null) {
209                index = ((index + offset) & 0x7FFFFFFF) % newCapacity;
210                offset = offset * 2 + 1;
211
212                if (offset == -1) {
213                    offset = 2;
214                }
215            }
216
217            newObjects[index] = o;
218        }
219        objects = (E[]) newObjects;
220        freecells = objects.length - elements;
221    }
222
223    private class CompactHashIterator implements Iterator<E> {
224
225        private int index;
226        private int lastReturned = -1;
227        private int expectedModCount;
228
229        public CompactHashIterator() {
230            for (index = 0; index < objects.length
231                    && (objects[index] == null
232                    || objects[index] == deletedObject); index++) {
233                ;
234            }
235            expectedModCount = modCount;
236        }
237
238        public boolean hasNext() {
239            return index < objects.length;
240        }
241
242        public E next() {
243            if (modCount != expectedModCount) {
244                throw new ConcurrentModificationException();
245            }
246            int length = objects.length;
247            if (index >= length) {
248                lastReturned = -2;
249                throw new NoSuchElementException();
250            }
251
252            lastReturned = index;
253            for (index += 1; index < length
254                    && (objects[index] == null
255                    || objects[index] == deletedObject); index++) {
256                ;
257            }
258            if (objects[lastReturned] == nullObject) {
259                return null;
260            } else {
261                return objects[lastReturned];
262            }
263        }
264
265        public void remove() {
266            if (modCount != expectedModCount) {
267                throw new ConcurrentModificationException();
268            }
269            if (lastReturned == -1 || lastReturned == -2) {
270                throw new IllegalStateException();
271            }
272            if (objects[lastReturned] != null && objects[lastReturned] != deletedObject) {
273                objects[lastReturned] = deletedObject;
274                elements--;
275                modCount++;
276                expectedModCount = modCount;
277            }
278        }
279    }
280}
