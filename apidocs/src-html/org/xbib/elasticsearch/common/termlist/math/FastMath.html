019/**
020 * Faster, more accurate, portable alternative to {@link Math} and
021 * {@link StrictMath} for large scale computation.
022 *
023 * FastMath is a drop-in replacement for both Math and StrictMath. This
024 * means that for any method in Math (say {@code Math.sin(x)} or
025 * {@code Math.cbrt(y)}), user can directly change the class and use the
026 * methods as is (using {@code FastMath.sin(x)} or {@code FastMath.cbrt(y)}
027 * in the previous example).
028 *
029 *
030 * FastMath speed is achieved by relying heavily on optimizing compilers
031 * to native code present in many JVMs today and use of large tables.
032 * The larger tables are lazily initialised on first use, so that the setup
033 * time does not penalise methods that don't need them.
034 *
035 *
036 * Note that FastMath is
037 * extensively used inside Apache Commons Math, so by calling some algorithms,
038 * the overhead when the the tables need to be intialised will occur
039 * regardless of the end-user calling FastMath methods directly or not.
040 * Performance figures for a specific JVM and hardware can be evaluated by
041 * running the FastMathTestPerformance tests in the test directory of the source
042 * distribution.
043 *
044 *
045 * FastMath accuracy should be mostly independent of the JVM as it relies only
046 * on IEEE-754 basic operations and on embedded tables. Almost all operations
047 * are accurate to about 0.5 ulp throughout the domain range. This statement,
048 * of course is only a rough global observed behavior, it is <em>not</em> a
049 * guarantee for <em>every</em> double numbers input (see William Kahan's <a
050 * href="http://en.wikipedia.org/wiki/Rounding#The_table-maker.27s_dilemma">Table
051 * Maker's Dilemma</a>).
052 *
053 *
054 * FastMath additionally implements the following methods not found in Math/StrictMath:
055 * <ul>
056 * <li>{@link #asinh(double)}</li>
057 * <li>{@link #acosh(double)}</li>
058 * <li>{@link #atanh(double)}</li>
059 * </ul>
060 * The following methods are found in Math/StrictMath since 1.6 only, they are provided
061 * by FastMath even in 1.5 Java virtual machines
062 * <ul>
063 * <li>{@link #copySign(double, double)}</li>
064 * <li>{@link #getExponent(double)}</li>
065 * <li>{@link #nextAfter(double,double)}</li>
066 * <li>{@link #nextUp(double)}</li>
067 * <li>{@link #scalb(double, int)}</li>
068 * <li>{@link #copySign(float, float)}</li>
069 * <li>{@link #getExponent(float)}</li>
070 * <li>{@link #nextAfter(float,double)}</li>
071 * <li>{@link #nextUp(float)}</li>
072 * <li>{@link #scalb(float, int)}</li>
073 * </ul>
074 *
075 */

350     * Get the high order bits from the mantissa.
351     * Equivalent to adding and subtracting HEX_40000 but also works for very large numbers

353     * @param d the value to split

356    private static double doubleHighPart(double d) {
357        if (d > -Precision.SAFE_MIN && d < Precision.SAFE_MIN){
358            return d; // These are un-normalised - don't try to convert
359        }

374    /** Compute the hyperbolic cosine of a number.
375     * @param x number on which evaluation is done
376     * @return hyperbolic cosine of x
377     */
378    public static double cosh(double x) {
379      if (x != x) {
380          return x;
381      }
382
383      // cosh[z] = (exp(z) + exp(-z))/2
384
385      // for numbers with magnitude 20 or so,
386      // exp(-z) can be ignored in comparison with exp(z)
387
388      if (x > 20.0) {
389          return exp(x)/2.0;
390      }
391
392      if (x < -20) {
393          return exp(-x)/2.0;
394      }
395
396      double hiPrec[] = new double[2];
397      if (x < 0.0) {
398          x = -x;
399      }
400      exp(x, 0.0, hiPrec);
401
402      double ya = hiPrec[0] + hiPrec[1];
403      double yb = -(ya - hiPrec[0] - hiPrec[1]);
404
405      double temp = ya * HEX_40000000;
406      double yaa = ya + temp - temp;
407      double yab = ya - yaa;
408
409      // recip = 1/y
410      double recip = 1.0/ya;
411      temp = recip * HEX_40000000;
412      double recipa = recip + temp - temp;
413      double recipb = recip - recipa;
414
415      // Correct for rounding in division
416      recipb += (1.0 - yaa*recipa - yaa*recipb - yab*recipa - yab*recipb) * recip;
417      // Account for yb
418      recipb += -yb * recip * recip;
419
420      // y = y + 1/y
421      temp = ya + recipa;
422      yb += -(temp - ya - recipa);
423      ya = temp;
424      temp = ya + recipb;
425      yb += -(temp - ya - recipb);
426      ya = temp;
427
428      double result = ya + yb;
429      result *= 0.5;
430      return result;
431    }

433    /** Compute the hyperbolic sine of a number.
434     * @param x number on which evaluation is done
435     * @return hyperbolic sine of x
436     */
437    public static double sinh(double x) {
438      boolean negate = false;
439      if (x != x) {
440          return x;
441      }
442
443      // sinh[z] = (exp(z) - exp(-z) / 2
444
445      // for values of z larger than about 20,
446      // exp(-z) can be ignored in comparison with exp(z)
447
448      if (x > 20.0) {
449          return exp(x)/2.0;
450      }
451
452      if (x < -20) {
453          return -exp(-x)/2.0;
454      }
455
456      if (x == 0) {
457          return x;
458      }
459
460      if (x < 0.0) {
461          x = -x;
462          negate = true;
463      }
464
465      double result;
466
467      if (x > 0.25) {
468          double hiPrec[] = new double[2];
469          exp(x, 0.0, hiPrec);
470
471          double ya = hiPrec[0] + hiPrec[1];
472          double yb = -(ya - hiPrec[0] - hiPrec[1]);
473
474          double temp = ya * HEX_40000000;
475          double yaa = ya + temp - temp;
476          double yab = ya - yaa;
477
478          // recip = 1/y
479          double recip = 1.0/ya;
480          temp = recip * HEX_40000000;
481          double recipa = recip + temp - temp;
482          double recipb = recip - recipa;
483
484          // Correct for rounding in division
485          recipb += (1.0 - yaa*recipa - yaa*recipb - yab*recipa - yab*recipb) * recip;
486          // Account for yb
487          recipb += -yb * recip * recip;
488
489          recipa = -recipa;
490          recipb = -recipb;
491
492          // y = y + 1/y
493          temp = ya + recipa;
494          yb += -(temp - ya - recipa);
495          ya = temp;
496          temp = ya + recipb;
497          yb += -(temp - ya - recipb);
498          ya = temp;
499
500          result = ya + yb;
501          result *= 0.5;
502      }
503      else {
504          double hiPrec[] = new double[2];
505          expm1(x, hiPrec);
506
507          double ya = hiPrec[0] + hiPrec[1];
508          double yb = -(ya - hiPrec[0] - hiPrec[1]);
509
510          /* Compute expm1(-x) = -expm1(x) / (expm1(x) + 1) */
511          double denom = 1.0 + ya;
512          double denomr = 1.0 / denom;
513          double denomb = -(denom - 1.0 - ya) + yb;
514          double ratio = ya * denomr;
515          double temp = ratio * HEX_40000000;
516          double ra = ratio + temp - temp;
517          double rb = ratio - ra;
518
519          temp = denom * HEX_40000000;
520          double za = denom + temp - temp;
521          double zb = denom - za;
522
523          rb += (ya - za*ra - za*rb - zb*ra - zb*rb) * denomr;
524
525          // Adjust for yb
526          rb += yb*denomr;                        // numerator
527          rb += -ya * denomb * denomr * denomr;   // denominator
528
529          // y = y - 1/y
530          temp = ya + ra;
531          yb += -(temp - ya - ra);
532          ya = temp;
533          temp = ya + rb;
534          yb += -(temp - ya - rb);
535          ya = temp;
536
537          result = ya + yb;
538          result *= 0.5;
539      }
540
541      if (negate) {
542          result = -result;
543      }
544
545      return result;
546    }

548    /** Compute the hyperbolic tangent of a number.
549     * @param x number on which evaluation is done
550     * @return hyperbolic tangent of x
551     */
552    public static double tanh(double x) {
553      boolean negate = false;
554
555      if (x != x) {
556          return x;
557      }
558
559      // tanh[z] = sinh[z] / cosh[z]
560      // = (exp(z) - exp(-z)) / (exp(z) + exp(-z))
561      // = (exp(2x) - 1) / (exp(2x) + 1)
562
563      // for magnitude > 20, sinh[z] == cosh[z] in double precision
564
565      if (x > 20.0) {
566          return 1.0;
567      }
568
569      if (x < -20) {
570          return -1.0;
571      }
572
573      if (x == 0) {
574          return x;
575      }
576
577      if (x < 0.0) {
578          x = -x;
579          negate = true;
580      }
581
582      double result;
583      if (x >= 0.5) {
584          double hiPrec[] = new double[2];
585          // tanh(x) = (exp(2x) - 1) / (exp(2x) + 1)
586          exp(x*2.0, 0.0, hiPrec);
587
588          double ya = hiPrec[0] + hiPrec[1];
589          double yb = -(ya - hiPrec[0] - hiPrec[1]);
590
591          /* Numerator */
592          double na = -1.0 + ya;
593          double nb = -(na + 1.0 - ya);
594          double temp = na + yb;
595          nb += -(temp - na - yb);
596          na = temp;
597
598          /* Denominator */
599          double da = 1.0 + ya;
600          double db = -(da - 1.0 - ya);
601          temp = da + yb;
602          db += -(temp - da - yb);
603          da = temp;
604
605          temp = da * HEX_40000000;
606          double daa = da + temp - temp;
607          double dab = da - daa;
608
609          // ratio = na/da
610          double ratio = na/da;
611          temp = ratio * HEX_40000000;
612          double ratioa = ratio + temp - temp;
613          double ratiob = ratio - ratioa;
614
615          // Correct for rounding in division
616          ratiob += (na - daa*ratioa - daa*ratiob - dab*ratioa - dab*ratiob) / da;
617
618          // Account for nb
619          ratiob += nb / da;
620          // Account for db
621          ratiob += -db * na / da / da;
622
623          result = ratioa + ratiob;
624      }
625      else {
626          double hiPrec[] = new double[2];
627          // tanh(x) = expm1(2x) / (expm1(2x) + 2)
628          expm1(x*2.0, hiPrec);
629
630          double ya = hiPrec[0] + hiPrec[1];
631          double yb = -(ya - hiPrec[0] - hiPrec[1]);
632
633          /* Numerator */
634          double na = ya;
635          double nb = yb;
636
637          /* Denominator */
638          double da = 2.0 + ya;
639          double db = -(da - 2.0 - ya);
640          double temp = da + yb;
641          db += -(temp - da - yb);
642          da = temp;
643
644          temp = da * HEX_40000000;
645          double daa = da + temp - temp;
646          double dab = da - daa;
647
648          // ratio = na/da
649          double ratio = na/da;
650          temp = ratio * HEX_40000000;
651          double ratioa = ratio + temp - temp;
652          double ratiob = ratio - ratioa;
653
654          // Correct for rounding in division
655          ratiob += (na - daa*ratioa - daa*ratiob - dab*ratioa - dab*ratiob) / da;
656
657          // Account for nb
658          ratiob += nb / da;
659          // Account for db
660          ratiob += -db * na / da / da;
661
662          result = ratioa + ratiob;
663      }
664
665      if (negate) {
666          result = -result;
667      }
668
669      return result;
670    }

782    /**
783     * Exponential function.
784     *
785     * Computes exp(x), function result is nearly rounded.   It will be correctly
786     * rounded to the theoretical value for 99.9% of input values, otherwise it will
787     * have a 1 UPL error.
788     *
789     * Method:
790     *    Lookup intVal = exp(int(x))
791     *    Lookup fracVal = exp(int(x-int(x) / 1024.0) * 1024.0 );
792     *    Compute z as the exponential of the remaining bits by a polynomial minus one
793     *    exp(x) = intVal * fracVal * (1 + z)
794     *
795     * Accuracy:
796     *    Calculation is done with 63 bits of precision, so result should be correctly
797     *    rounded for 99.9% of input values, with less than 1 ULP error otherwise.
798     *
799     * @param x   a double
800     * @return double e<sup>x</sup>
801     */
802    public static double exp(double x) {
803        return exp(x, 0.0, null);
804    }

806    /**
807     * Internal helper method for exponential function.
808     * @param x original argument of the exponential function
809     * @param extra extra bits of precision on input (To Be Confirmed)
810     * @param hiPrec extra bits of precision on output (To Be Confirmed)
811     * @return exp(x)
812     */
813    private static double exp(double x, double extra, double[] hiPrec) {
814        double intPartA;
815        double intPartB;
816        int intVal;
817
818        /* Lookup exp(floor(x)).
819         * intPartA will have the upper 22 bits, intPartB will have the lower
820         * 52 bits.
821         */
822        if (x < 0.0) {
823            intVal = (int) -x;
824
825            if (intVal > 746) {
826                if (hiPrec != null) {
827                    hiPrec[0] = 0.0;
828                    hiPrec[1] = 0.0;
829                }
830                return 0.0;
831            }
832
833            if (intVal > 709) {
834                /* This will produce a subnormal output */
835                final double result = exp(x+40.19140625, extra, hiPrec) / 285040095144011776.0;
836                if (hiPrec != null) {
837                    hiPrec[0] /= 285040095144011776.0;
838                    hiPrec[1] /= 285040095144011776.0;
839                }
840                return result;
841            }
842
843            if (intVal == 709) {
844                /* exp(1.494140625) is nearly a machine number... */
845                final double result = exp(x+1.494140625, extra, hiPrec) / 4.455505956692756620;
846                if (hiPrec != null) {
847                    hiPrec[0] /= 4.455505956692756620;
848                    hiPrec[1] /= 4.455505956692756620;
849                }
850                return result;
851            }
852
853            intVal++;
854
855            intPartA = ExpIntTable.EXP_INT_TABLE_A[EXP_INT_TABLE_MAX_INDEX-intVal];
856            intPartB = ExpIntTable.EXP_INT_TABLE_B[EXP_INT_TABLE_MAX_INDEX-intVal];
857
858            intVal = -intVal;
859        } else {
860            intVal = (int) x;
861
862            if (intVal > 709) {
863                if (hiPrec != null) {
864                    hiPrec[0] = Double.POSITIVE_INFINITY;
865                    hiPrec[1] = 0.0;
866                }
867                return Double.POSITIVE_INFINITY;
868            }
869
870            intPartA = ExpIntTable.EXP_INT_TABLE_A[EXP_INT_TABLE_MAX_INDEX+intVal];
871            intPartB = ExpIntTable.EXP_INT_TABLE_B[EXP_INT_TABLE_MAX_INDEX+intVal];
872        }
873
874        /* Get the fractional part of x, find the greatest multiple of 2^-10 less than
875         * x and look up the exp function of it.
876         * fracPartA will have the upper 22 bits, fracPartB the lower 52 bits.
877         */
878        final int intFrac = (int) ((x - intVal) * 1024.0);
879        final double fracPartA = ExpFracTable.EXP_FRAC_TABLE_A[intFrac];
880        final double fracPartB = ExpFracTable.EXP_FRAC_TABLE_B[intFrac];
881
882        /* epsilon is the difference in x from the nearest multiple of 2^-10.  It
883         * has a value in the range 0 <= epsilon < 2^-10.
884         * Do the subtraction from x as the last step to avoid possible loss of percison.
885         */
886        final double epsilon = x - (intVal + intFrac / 1024.0);
887
888        /* Compute z = exp(epsilon) - 1.0 via a minimax polynomial.  z has
889       full double precision (52 bits).  Since z < 2^-10, we will have
890       62 bits of precision when combined with the contant 1.  This will be
891       used in the last addition below to get proper rounding. */
892
893        /* Remez generated polynomial.  Converges on the interval [0, 2^-10], error
894       is less than 0.5 ULP */
895        double z = 0.04168701738764507;
896        z = z * epsilon + 0.1666666505023083;
897        z = z * epsilon + 0.5000000000042687;
898        z = z * epsilon + 1.0;
899        z = z * epsilon + -3.940510424527919E-20;
900
901        /* Compute (intPartA+intPartB) * (fracPartA+fracPartB) by binomial
902       expansion.
903       tempA is exact since intPartA and intPartB only have 22 bits each.
904       tempB will have 52 bits of precision.
905         */
906        double tempA = intPartA * fracPartA;
907        double tempB = intPartA * fracPartB + intPartB * fracPartA + intPartB * fracPartB;
908
909        /* Compute the result.  (1+z)(tempA+tempB).  Order of operations is
910       important.  For accuracy add by increasing size.  tempA is exact and
911       much larger than the others.  If there are extra bits specified from the
912       pow() function, use them. */
913        final double tempC = tempB + tempA;
914        final double result;
915        if (extra != 0.0) {
916            result = tempC*extra*z + tempC*extra + tempC*z + tempB + tempA;
917        } else {
918            result = tempC*z + tempB + tempA;
919        }
920
921        if (hiPrec != null) {
922            // If requesting high precision
923            hiPrec[0] = tempA;
924            hiPrec[1] = tempC*extra*z + tempC*extra + tempC*z + tempB;
925        }
926
927        return result;
928    }

942     */
943    private static double expm1(double x, double hiPrecOut[]) {

945            return x;
946        }

949            // If not between +/- 1.0

955            } else {

957                double rb = -(ra + 1.0 - hiPrec[0]);
958                rb += hiPrec[1];
959                return ra + rb;
960            }
961        }
962
963        double baseA;
964        double baseB;
965        double epsilon;
966        boolean negative = false;
967
968        if (x < 0.0) {
969            x = -x;
970            negative = true;
971        }
972
973        {
974            int intFrac = (int) (x * 1024.0);
975            double tempA = ExpFracTable.EXP_FRAC_TABLE_A[intFrac] - 1.0;
976            double tempB = ExpFracTable.EXP_FRAC_TABLE_B[intFrac];
977
978            double temp = tempA + tempB;
979            tempB = -(temp - tempA - tempB);
980            tempA = temp;
981
982            temp = tempA * HEX_40000000;
983            baseA = tempA + temp - temp;
984            baseB = tempB + (tempA - baseA);
985
986            epsilon = x - intFrac/1024.0;
987        }
988
989
990        /* Compute expm1(epsilon) */
991        double zb = 0.008336750013465571;
992        zb = zb * epsilon + 0.041666663879186654;
993        zb = zb * epsilon + 0.16666666666745392;
994        zb = zb * epsilon + 0.49999999999999994;
995        zb = zb * epsilon;
996        zb = zb * epsilon;
997
998        double za = epsilon;
999        double temp = za + zb;
1000        zb = -(temp - za - zb);
1001        za = temp;
1002
1003        temp = za * HEX_40000000;
1004        temp = za + temp - temp;
1005        zb += za - temp;
1006        za = temp;
1007
1008        /* Combine the parts.   expm1(a+b) = expm1(a) + expm1(b) + expm1(a)*expm1(b) */
1009        double ya = za * baseA;
1010        //double yb = za*baseB + zb*baseA + zb*baseB;
1011        temp = ya + za * baseB;
1012        double yb = -(temp - ya - za * baseB);
1013        ya = temp;
1014
1015        temp = ya + zb * baseA;
1016        yb += -(temp - ya - zb * baseA);
1017        ya = temp;
1018
1019        temp = ya + zb * baseB;
1020        yb += -(temp - ya - zb*baseB);
1021        ya = temp;
1022
1023        //ya = ya + za + baseA;
1024        //yb = yb + zb + baseB;
1025        temp = ya + baseA;
1026        yb += -(temp - baseA - ya);
1027        ya = temp;
1028
1029        temp = ya + za;
1030        //yb += (ya > za) ? -(temp - ya - za) : -(temp - za - ya);
1031        yb += -(temp - ya - za);
1032        ya = temp;
1033
1034        temp = ya + baseB;
1035        //yb += (ya > baseB) ? -(temp - ya - baseB) : -(temp - baseB - ya);
1036        yb += -(temp - ya - baseB);
1037        ya = temp;
1038
1039        temp = ya + zb;
1040        //yb += (ya > zb) ? -(temp - ya - zb) : -(temp - zb - ya);
1041        yb += -(temp - ya - zb);
1042        ya = temp;
1043
1044        if (negative) {
1045            /* Compute expm1(-x) = -expm1(x) / (expm1(x) + 1) */
1046            double denom = 1.0 + ya;
1047            double denomr = 1.0 / denom;
1048            double denomb = -(denom - 1.0 - ya) + yb;
1049            double ratio = ya * denomr;
1050            temp = ratio * HEX_40000000;
1051            final double ra = ratio + temp - temp;
1052            double rb = ratio - ra;
1053
1054            temp = denom * HEX_40000000;
1055            za = denom + temp - temp;
1056            zb = denom - za;
1057
1058            rb += (ya - za * ra - za * rb - zb * ra - zb * rb) * denomr;
1059
1060            // f(x) = x/1+x
1061            // Compute f'(x)
1062            // Product rule:  d(uv) = du*v + u*dv
1063            // Chain rule:  d(f(g(x)) = f'(g(x))*f(g'(x))
1064            // d(1/x) = -1/(x*x)
1065            // d(1/1+x) = -1/( (1+x)^2) *  1 =  -1/((1+x)*(1+x))
1066            // d(x/1+x) = -x/((1+x)(1+x)) + 1/1+x = 1 / ((1+x)(1+x))
1067
1068            // Adjust for yb
1069            rb += yb * denomr;                      // numerator
1070            rb += -ya * denomb * denomr * denomr;   // denominator
1071
1072            // negate
1073            ya = -ra;
1074            yb = -rb;
1075        }
1076
1077        if (hiPrecOut != null) {
1078            hiPrecOut[0] = ya;
1079            hiPrecOut[1] = yb;
1080        }
1081
1082        return ya + yb;
1083    }

1179                    /* Split y = a */

1183                }

1186                aa = ya * xa;
1187                ab = ya * xb + yb * xa + yb * xb;
1188                /* split, so now y = a */
1189                tmp = aa * HEX_40000000;

1201    double epsilon = x / Double.longBitsToDouble(bits & 0xfffffc0000000000L);
1202
1203    epsilon -= 1.0;
1204         */
1205
1206        // y is the most significant 10 bits of the mantissa
1207        //double y = Double.longBitsToDouble(bits & 0xfffffc0000000000L);
1208        //double epsilon = (x - y) / y;
1209        double epsilon = (bits & 0x3ffffffffffL) / (TWO_POWER_52 + (bits & 0x000ffc0000000000L));
1210
1211        double lnza = 0.0;
1212        double lnzb = 0.0;
1213
1214        if (hiPrec != null) {
1215            /* split epsilon -> x */
1216            double tmp = epsilon * HEX_40000000;
1217            double aa = epsilon + tmp - tmp;
1218            double ab = epsilon - aa;
1219            double xa = aa;
1220            double xb = ab;
1221
1222            /* Need a more accurate epsilon, so adjust the division. */
1223            double numer = bits & 0x3ffffffffffL;
1224            double denom = TWO_POWER_52 + (bits & 0x000ffc0000000000L);
1225            aa = numer - xa*denom - xb * denom;
1226            xb += aa / denom;
1227
1228            /* Remez polynomial evaluation */
1229            double ya = LN_HI_PREC_COEF[LN_HI_PREC_COEF.length-1][0];
1230            double yb = LN_HI_PREC_COEF[LN_HI_PREC_COEF.length-1][1];
1231
1232            for (int i = LN_HI_PREC_COEF.length - 2; i >= 0; i--) {
1233                /* Multiply a = y * x */
1234                aa = ya * xa;
1235                ab = ya * xb + yb * xa + yb * xb;
1236                /* split, so now y = a */
1237                tmp = aa * HEX_40000000;
1238                ya = aa + tmp - tmp;
1239                yb = aa - ya + ab;
1240
1241                /* Add  a = y + lnHiPrecCoef */
1242                aa = ya + LN_HI_PREC_COEF[i][0];
1243                ab = yb + LN_HI_PREC_COEF[i][1];
1244                /* Split y = a */
1245                tmp = aa * HEX_40000000;
1246                ya = aa + tmp - tmp;
1247                yb = aa - ya + ab;
1248            }
1249
1250            /* Multiply a = y * x */
1251            aa = ya * xa;
1252            ab = ya * xb + yb * xa + yb * xb;
1253
1254            /* split, so now lnz = a */
1255            /*
1256      tmp = aa * 1073741824.0;
1257      lnza = aa + tmp - tmp;
1258      lnzb = aa - lnza + ab;
1259             */
1260            lnza = aa + ab;
1261            lnzb = -(lnza - aa - ab);
1262        } else {
1263            /* High precision not required.  Eval Remez polynomial
1264         using standard double precision */
1265            lnza = -0.16624882440418567;
1266            lnza = lnza * epsilon + 0.19999954120254515;
1267            lnza = lnza * epsilon + -0.2499999997677497;
1268            lnza = lnza * epsilon + 0.3333333333332802;
1269            lnza = lnza * epsilon + -0.5;
1270            lnza = lnza * epsilon + 1.0;
1271            lnza = lnza * epsilon;
1272        }
1273
1274        /* Relative sizes:
1275         * lnzb     [0, 2.33E-10]
1276         * lnm[1]   [0, 1.17E-7]
1277         * ln2B*exp [0, 1.12E-4]
1278         * lnza      [0, 9.7E-4]
1279         * lnm[0]   [0, 0.692]
1280         * ln2A*exp [0, 709]
1281         */
1282
1283        /* Compute the following sum:
1284         * lnzb + lnm[1] + ln2B*exp + lnza + lnm[0] + ln2A*exp;
1285         */
1286
1287        //return lnzb + lnm[1] + ln2B*exp + lnza + lnm[0] + ln2A*exp;
1288        double a = LN_2_A*exp;
1289        double b = 0.0;
1290        double c = a+lnm[0];
1291        double d = -(c-a-lnm[0]);
1292        a = c;
1293        b = b + d;
1294
1295        c = a + lnza;
1296        d = -(c - a - lnza);
1297        a = c;
1298        b = b + d;
1299
1300        c = a + LN_2_B*exp;
1301        d = -(c - a - LN_2_B*exp);
1302        a = c;
1303        b = b + d;
1304
1305        c = a + lnm[1];
1306        d = -(c - a - lnm[1]);
1307        a = c;
1308        b = b + d;
1309
1310        c = a + lnzb;
1311        d = -(c - a - lnzb);
1312        a = c;
1313        b = b + d;
1314
1315        if (hiPrec != null) {
1316            hiPrec[0] = a;
1317            hiPrec[1] = b;
1318        }
1319
1320        return a + b;
1321    }

1323    /** Compute log(1 + x).
1324     * @param x a number
1325     * @return log(1 + x)
1326     */
1327    public static double log1p(final double x) {
1328
1329        if (x == -1) {
1330            return x/0.0;   // -Infinity
1331        }
1332
1333        if (x > 0 && 1/x == 0) { // x = Infinity
1334            return x;
1335        }
1336
1337        if (x>1e-6 || x<-1e-6) {
1338            double xpa = 1.0 + x;
1339            double xpb = -(xpa - 1.0 - x);
1340
1341            double hiPrec[] = new double[2];
1342
1343            final double lores = log(xpa, hiPrec);
1344            if (Double.isInfinite(lores)){ // don't allow this to be converted to NaN
1345                return lores;
1346            }
1347
1348            /* Do a taylor series expansion around xpa */
1349            /* f(x+y) = f(x) + f'(x)*y + f''(x)/2 y^2 */
1350            double fx1 = xpb/xpa;
1351
1352            double epsilon = 0.5 * fx1 + 1.0;
1353            epsilon = epsilon * fx1;
1354
1355            return epsilon + hiPrec[1] + hiPrec[0];
1356        }
1357
1358        /* Value is small |x| < 1e6, do a Taylor series centered on 1.0 */
1359        double y = x * F_1_3 - F_1_2;
1360        y = y * x + 1.0;
1361        y = y * x;
1362
1363        return y;
1364    }

1408    /**
1409     * Power function.  Compute x^y.
1410     *
1411     * @param x   a double
1412     * @param y   a double
1413     * @return double
1414     */
1415    public static double pow(double x, double y) {
1416        final double lns[] = new double[2];
1417
1418        if (y == 0.0) {
1419            return 1.0;
1420        }
1421
1422        if (x != x) { // X is NaN
1423            return x;
1424        }
1425
1426
1427        if (x == 0) {
1428            long bits = Double.doubleToLongBits(x);
1429            if ((bits & 0x8000000000000000L) != 0) {
1430                // -zero
1431                long yi = (long) y;
1432
1433                if (y < 0 && y == yi && (yi & 1) == 1) {
1434                    return Double.NEGATIVE_INFINITY;
1435                }
1436
1437                if (y > 0 && y == yi && (yi & 1) == 1) {
1438                    return -0.0;
1439                }
1440            }
1441
1442            if (y < 0) {
1443                return Double.POSITIVE_INFINITY;
1444            }
1445            if (y > 0) {
1446                return 0.0;
1447            }
1448
1449            return Double.NaN;
1450        }
1451
1452        if (x == Double.POSITIVE_INFINITY) {
1453            if (y != y) { // y is NaN
1454                return y;
1455            }
1456            if (y < 0.0) {
1457                return 0.0;
1458            } else {
1459                return Double.POSITIVE_INFINITY;
1460            }
1461        }
1462
1463        if (y == Double.POSITIVE_INFINITY) {
1464            if (x * x == 1.0) {
1465                return Double.NaN;
1466            }
1467
1468            if (x * x > 1.0) {
1469                return Double.POSITIVE_INFINITY;
1470            } else {
1471                return 0.0;
1472            }
1473        }
1474
1475        if (x == Double.NEGATIVE_INFINITY) {
1476            if (y != y) { // y is NaN
1477                return y;
1478            }
1479
1480            if (y < 0) {
1481                long yi = (long) y;
1482                if (y == yi && (yi & 1) == 1) {
1483                    return -0.0;
1484                }
1485
1486                return 0.0;
1487            }
1488
1489            if (y > 0)  {
1490                long yi = (long) y;
1491                if (y == yi && (yi & 1) == 1) {
1492                    return Double.NEGATIVE_INFINITY;
1493                }
1494
1495                return Double.POSITIVE_INFINITY;
1496            }
1497        }
1498
1499        if (y == Double.NEGATIVE_INFINITY) {
1500
1501            if (x * x == 1.0) {
1502                return Double.NaN;
1503            }
1504
1505            if (x * x < 1.0) {
1506                return Double.POSITIVE_INFINITY;
1507            } else {
1508                return 0.0;
1509            }
1510        }
1511
1512        /* Handle special case x<0 */
1513        if (x < 0) {
1514            // y is an even integer in this case
1515            if (y >= TWO_POWER_52 || y <= -TWO_POWER_52) {
1516                return pow(-x, y);
1517            }
1518
1519            if (y == (long) y) {
1520                // If y is an integer
1521                return ((long)y & 1) == 0 ? pow(-x, y) : -pow(-x, y);
1522            } else {
1523                return Double.NaN;
1524            }
1525        }
1526
1527        /* Split y into ya and yb such that y = ya+yb */
1528        double ya;
1529        double yb;
1530        if (y < 8e298 && y > -8e298) {
1531            double tmp1 = y * HEX_40000000;
1532            ya = y + tmp1 - tmp1;
1533            yb = y - ya;
1534        } else {
1535            double tmp1 = y * 9.31322574615478515625E-10;
1536            double tmp2 = tmp1 * 9.31322574615478515625E-10;
1537            ya = (tmp1 + tmp2 - tmp1) * HEX_40000000 * HEX_40000000;
1538            yb = y - ya;
1539        }
1540
1541        /* Compute ln(x) */
1542        final double lores = log(x, lns);
1543        if (Double.isInfinite(lores)){ // don't allow this to be converted to NaN
1544            return lores;
1545        }
1546
1547        double lna = lns[0];
1548        double lnb = lns[1];
1549
1550        /* resplit lns */
1551        double tmp1 = lna * HEX_40000000;
1552        double tmp2 = lna + tmp1 - tmp1;
1553        lnb += lna - tmp2;
1554        lna = tmp2;
1555
1556        // y*ln(x) = (aa+ab)
1557        final double aa = lna * ya;
1558        final double ab = lna * yb + lnb * ya + lnb * yb;
1559
1560        lna = aa+ab;
1561        lnb = -(lna - aa - ab);
1562
1563        double z = 1.0 / 120.0;
1564        z = z * lnb + (1.0 / 24.0);
1565        z = z * lnb + (1.0 / 6.0);
1566        z = z * lnb + 0.5;
1567        z = z * lnb + 1.0;
1568        z = z * lnb;
1569
1570        final double result = exp(lna, z, null);
1571        //result = result + result * z;
1572        return result;
1573    }

1730            a = c;

1735

1825        b = b + d;
1826
1827        b = b + sintA*cosEpsB + costA*sinEpsB;
1828        b = b + sintB + costB*sinEpsA + sintB*cosEpsB + costB*sinEpsB;
1829
1830        double sina = a + b;
1831        double sinb = -(sina - a - b);
1832
1833        // Compute cosine
1834
1835        a = b = c = d = 0.0;
1836
1837        t = costA*cosEpsA;
1838        c = a + t;
1839        d = -(c - a - t);
1840        a = c;
1841        b = b + d;
1842
1843        t = -sintA*sinEpsA;
1844        c = a + t;
1845        d = -(c - a - t);
1846        a = c;
1847        b = b + d;
1848
1849        b = b + costB*cosEpsA + costA*cosEpsB + costB*cosEpsB;
1850        b = b - (sintB*sinEpsA + sintA*sinEpsB + sintB*sinEpsB);

1852        double cosa = a + b;
1853        double cosb = -(cosa - a - b);
1854
1855        if (cotanFlag) {
1856            double tmp;
1857            tmp = cosa; cosa = sina; sina = tmp;
1858            tmp = cosb; cosb = sinb; sinb = tmp;
1859        }
1860

1862        /* estimate and correct, compute 1.0/(cosa+cosb) */
1863        /*
1864    double est = (sina+sinb)/(cosa+cosb);
1865    double err = (sina - cosa*est) + (sinb - cosb*est);
1866    est += err/(cosa+cosb);
1867    err = (sina - cosa*est) + (sinb - cosb*est);
1868         */
1869
1870        // f(x) = 1/x,   f'(x) = -1/x^2
1871
1872        double est = sina/cosa;
1873
1874        /* Split the estimate to get more accurate read on division rounding */
1875        temp = est * HEX_40000000;
1876        double esta = (est + temp) - temp;
1877        double estb =  est - esta;
1878
1879        temp = cosa * HEX_40000000;
1880        double cosaa = (cosa + temp) - temp;
1881        double cosab =  cosa - cosaa;
1882
1883        //double err = (sina - est*cosa)/cosa;  // Correction for division rounding
1884        double err = (sina - esta*cosaa - esta*cosab - estb*cosaa - estb*cosab)/cosa;  // Correction for division rounding
1885        err += sinb/cosa;                     // Change in est due to sinb
1886        err += -sina * cosb / cosa / cosa;    // Change in est due to cosb
1887
1888        if (xb != 0.0) {
1889            // tan' = 1 + tan^2      cot' = -(1 + cot^2)
1890            // Approximate impact of xb
1891            double xbadj = xb + est*est*xb;
1892            if (cotanFlag) {
1893                xbadj = -xbadj;
1894            }
1895
1896            err += xbadj;
1897        }
1898
1899        return est+err;

1902    /** Reduce the input argument using the Payne and Hanek method.
1903     *  This is good for all inputs 0.0 < x < inf
1904     *  Output is remainder after dividing by PI/2
1905     *  The result array should contain 3 numbers.
1906     *  result[0] is the integer portion, so mod 4 this gives the quadrant.
1907     *  result[1] is the upper bits of the remainder
1908     *  result[2] is the lower bits of the remainder
1909     *
1910     * @param x number to reduce
1911     * @param result placeholder where to put the result
1912     */
1913    private static void reducePayneHanek(double x, double result[])
1914    {
1915        /* Convert input double to bits */
1916        long inbits = Double.doubleToLongBits(x);
1917        int exponent = (int) ((inbits >> 52) & 0x7ff) - 1023;
1918
1919        /* Convert to fixed point representation */
1920        inbits &= 0x000fffffffffffffL;
1921        inbits |= 0x0010000000000000L;
1922
1923        /* Normalize input to be between 0.5 and 1.0 */
1924        exponent++;
1925        inbits <<= 11;
1926
1927        /* Based on the exponent, get a shifted copy of recip2pi */
1928        long shpi0;
1929        long shpiA;
1930        long shpiB;
1931        int idx = exponent >> 6;
1932        int shift = exponent - (idx << 6);
1933
1934        if (shift != 0) {
1935            shpi0 = (idx == 0) ? 0 : (RECIP_2PI[idx-1] << shift);
1936            shpi0 |= RECIP_2PI[idx] >>> (64-shift);
1937            shpiA = (RECIP_2PI[idx] << shift) | (RECIP_2PI[idx+1] >>> (64-shift));
1938            shpiB = (RECIP_2PI[idx+1] << shift) | (RECIP_2PI[idx+2] >>> (64-shift));
1939        } else {
1940            shpi0 = (idx == 0) ? 0 : RECIP_2PI[idx-1];
1941            shpiA = RECIP_2PI[idx];
1942            shpiB = RECIP_2PI[idx+1];
1943        }
1944
1945        /* Multiply input by shpiA */
1946        long a = inbits >>> 32;
1947        long b = inbits & 0xffffffffL;
1948
1949        long c = shpiA >>> 32;
1950        long d = shpiA & 0xffffffffL;
1951
1952        long ac = a * c;
1953        long bd = b * d;
1954        long bc = b * c;
1955        long ad = a * d;
1956
1957        long prodB = bd + (ad << 32);
1958        long prodA = ac + (ad >>> 32);
1959
1960        boolean bita = (bd & 0x8000000000000000L) != 0;
1961        boolean bitb = (ad & 0x80000000L ) != 0;
1962        boolean bitsum = (prodB & 0x8000000000000000L) != 0;
1963
1964        /* Carry */
1965        if ( (bita && bitb) ||
1966                ((bita || bitb) && !bitsum) ) {
1967            prodA++;
1968        }
1969
1970        bita = (prodB & 0x8000000000000000L) != 0;
1971        bitb = (bc & 0x80000000L ) != 0;
1972
1973        prodB = prodB + (bc << 32);
1974        prodA = prodA + (bc >>> 32);
1975
1976        bitsum = (prodB & 0x8000000000000000L) != 0;
1977
1978        /* Carry */
1979        if ( (bita && bitb) ||
1980                ((bita || bitb) && !bitsum) ) {
1981            prodA++;
1982        }
1983
1984        /* Multiply input by shpiB */
1985        c = shpiB >>> 32;
1986        d = shpiB & 0xffffffffL;
1987        ac = a * c;
1988        bc = b * c;
1989        ad = a * d;
1990
1991        /* Collect terms */
1992        ac = ac + ((bc + ad) >>> 32);
1993
1994        bita = (prodB & 0x8000000000000000L) != 0;
1995        bitb = (ac & 0x8000000000000000L ) != 0;
1996        prodB += ac;
1997        bitsum = (prodB & 0x8000000000000000L) != 0;
1998        /* Carry */
1999        if ( (bita && bitb) ||
2000                ((bita || bitb) && !bitsum) ) {
2001            prodA++;
2002        }
2003
2004        /* Multiply by shpi0 */
2005        c = shpi0 >>> 32;
2006        d = shpi0 & 0xffffffffL;
2007
2008        bd = b * d;
2009        bc = b * c;
2010        ad = a * d;
2011
2012        prodA += bd + ((bc + ad) << 32);
2013
2014        /*
2015         * prodA, prodB now contain the remainder as a fraction of PI.  We want this as a fraction of
2016         * PI/2, so use the following steps:
2017         * 1.) multiply by 4.
2018         * 2.) do a fixed point muliply by PI/4.
2019         * 3.) Convert to floating point.
2020         * 4.) Multiply by 2
2021         */
2022
2023        /* This identifies the quadrant */
2024        int intPart = (int)(prodA >>> 62);
2025
2026        /* Multiply by 4 */
2027        prodA <<= 2;
2028        prodA |= prodB >>> 62;
2029        prodB <<= 2;
2030
2031        /* Multiply by PI/4 */
2032        a = prodA >>> 32;
2033        b = prodA & 0xffffffffL;
2034
2035        c = PI_O_4_BITS[0] >>> 32;
2036        d = PI_O_4_BITS[0] & 0xffffffffL;
2037
2038        ac = a * c;
2039        bd = b * d;
2040        bc = b * c;
2041        ad = a * d;
2042
2043        long prod2B = bd + (ad << 32);
2044        long prod2A = ac + (ad >>> 32);
2045
2046        bita = (bd & 0x8000000000000000L) != 0;
2047        bitb = (ad & 0x80000000L ) != 0;
2048        bitsum = (prod2B & 0x8000000000000000L) != 0;
2049
2050        /* Carry */
2051        if ( (bita && bitb) ||
2052                ((bita || bitb) && !bitsum) ) {
2053            prod2A++;
2054        }
2055
2056        bita = (prod2B & 0x8000000000000000L) != 0;
2057        bitb = (bc & 0x80000000L ) != 0;
2058
2059        prod2B = prod2B + (bc << 32);
2060        prod2A = prod2A + (bc >>> 32);
2061
2062        bitsum = (prod2B & 0x8000000000000000L) != 0;
2063
2064        /* Carry */
2065        if ( (bita && bitb) ||
2066                ((bita || bitb) && !bitsum) ) {
2067            prod2A++;
2068        }
2069
2070        /* Multiply input by pio4bits[1] */
2071        c = PI_O_4_BITS[1] >>> 32;
2072        d = PI_O_4_BITS[1] & 0xffffffffL;
2073        ac = a * c;
2074        bc = b * c;
2075        ad = a * d;
2076
2077        /* Collect terms */
2078        ac = ac + ((bc + ad) >>> 32);
2079
2080        bita = (prod2B & 0x8000000000000000L) != 0;
2081        bitb = (ac & 0x8000000000000000L ) != 0;
2082        prod2B += ac;
2083        bitsum = (prod2B & 0x8000000000000000L) != 0;
2084        /* Carry */
2085        if ( (bita && bitb) ||
2086                ((bita || bitb) && !bitsum) ) {
2087            prod2A++;
2088        }
2089
2090        /* Multiply inputB by pio4bits[0] */
2091        a = prodB >>> 32;
2092        b = prodB & 0xffffffffL;
2093        c = PI_O_4_BITS[0] >>> 32;
2094        d = PI_O_4_BITS[0] & 0xffffffffL;
2095        ac = a * c;
2096        bc = b * c;
2097        ad = a * d;
2098
2099        /* Collect terms */
2100        ac = ac + ((bc + ad) >>> 32);
2101
2102        bita = (prod2B & 0x8000000000000000L) != 0;
2103        bitb = (ac & 0x8000000000000000L ) != 0;
2104        prod2B += ac;
2105        bitsum = (prod2B & 0x8000000000000000L) != 0;
2106        /* Carry */
2107        if ( (bita && bitb) ||
2108                ((bita || bitb) && !bitsum) ) {
2109            prod2A++;
2110        }
2111
2112        /* Convert to double */
2113        double tmpA = (prod2A >>> 12) / TWO_POWER_52;  // High order 52 bits
2114        double tmpB = (((prod2A & 0xfffL) << 40) + (prod2B >>> 24)) / TWO_POWER_52 / TWO_POWER_52; // Low bits
2115
2116        double sumA = tmpA + tmpB;
2117        double sumB = -(sumA - tmpA - tmpB);
2118
2119        /* Multiply by PI/2 and return */
2120        result[0] = intPart;
2121        result[1] = sumA * 2.0;
2122        result[2] = sumB * 2.0;
2123    }
