017package org.xbib.elasticsearch.common.termlist.math;
018
019import org.elasticsearch.common.io.stream.StreamInput;
020import org.elasticsearch.common.io.stream.StreamOutput;
021
022import java.io.IOException;
023
024/**
025 * Computes the arithmetic mean of a set of values. Uses the definitional
026 * formula:
027 *
028 * mean = sum(x_i) / n
029 *
030 * where <code>n</code> is the number of observations.
031 *
032 * When {@link #increment(double)} is used to add data incrementally from a
033 * stream of (unstored) values, the value of the statistic that
034 * {@link #getResult()} returns is computed using the following recursive
035 * updating algorithm:
036 * <ol>
037 * <li>Initialize <code>m = </code> the first value</li>
038 * <li>For each additional value, update using
039 *   <code>m = m + (new value - m) / (number of observations)</code></li>
040 * </ol>
041 * If {@link #evaluate(double[])} is used to compute the mean of an array
042 * of stored values, a two-pass, corrected algorithm is used, starting with
043 * the definitional formula computed using the array of stored values and then
044 * correcting this by adding the mean deviation of the data values from the
045 * arithmetic mean. See, e.g. "Comparison of Several Algorithms for Computing
046 * Sample Means and Variances," Robert F. Ling, Journal of the American
047 * Statistical Association, Vol. 69, No. 348 (Dec., 1974), pp. 859-866.
048 *
049 * Returns <code>Double.NaN</code> if the dataset is empty.
050 *
051 * <strong>Note that this implementation is not synchronized.</strong> If
052 * multiple threads access an instance of this class concurrently, and at least
053 * one of the threads invokes the <code>increment()</code> or
054 * <code>clear()</code> method, it must be synchronized externally.
055 */
056public class Mean extends AbstractStorelessUnivariateStatistic {
057
058    /** First moment on which this statistic is based. */
059    protected FirstMoment moment;
060
061    /**
062     * Determines whether or not this statistic can be incremented or cleared.
063     *
064     * Statistics based on (constructed from) external moments cannot
065     * be incremented or cleared.
066     */
067    protected boolean incMoment;
068
069    /** Constructs a Mean. */
070    public Mean() {
071        incMoment = true;
072        moment = new FirstMoment();
073    }
074
075    /**
076     * Constructs a Mean with an External Moment.
077     *
078     * @param m1 the moment
079     */
080    public Mean(final FirstMoment m1) {
081        this.moment = m1;
082        incMoment = false;
083    }
084
085    @Override
086    public void increment(final double d) {
087        if (incMoment) {
088            moment.increment(d);
089        }
090    }
091
092    @Override
093    public void clear() {
094        if (incMoment) {
095            moment.clear();
096        }
097    }
098
099    @Override
100    public double getResult() {
101        return moment.m1;
102    }
103
104    public long getN() {
105        return moment.getN();
106    }
107
108    /**
109     * Returns the arithmetic mean of the entries in the specified portion of
110     * the input array, or <code>Double.NaN</code> if the designated subarray
111     * is empty.
112     *
113     * Throws <code>IllegalArgumentException</code> if the array is null.
114     *
115     * See {@link Mean} for details on the computing algorithm.
116     *
117     * @param values the input array
118     * @param begin index of the first array element to include
119     * @param length the number of elements to include
120     * @return the mean of the values or Double.NaN if length = 0
121     * @throws IllegalArgumentException if the array is null or the array index
122     *  parameters are not valid
123     */
124    @Override
125    public double evaluate(final double[] values,final int begin, final int length) {
126        if (test(values, begin, length)) {
127            Sum sum = new Sum();
128            // Compute initial estimate using definitional formula
129            double xbar = sum.evaluate(values, begin, length) / (double) length;
130            // Compute correction factor in second pass
131            double correction = 0;
132            for (int i = begin; i < begin + length; i++) {
133                correction += values[i] - xbar;
134            }
135            return xbar + (correction/ (double) length);
136        }
137        return Double.NaN;
138    }
139
140    /**
141     * Returns the weighted arithmetic mean of the entries in the specified portion of
142     * the input array, or <code>Double.NaN</code> if the designated subarray
143     * is empty.
144     *
145     * Throws <code>IllegalArgumentException</code> if either array is null.
146     *
147     * See {@link Mean} for details on the computing algorithm. The two-pass algorithm
148     * described above is used here, with weights applied in computing both the original
149     * estimate and the correction factor.
150     *
151     * Throws <code>IllegalArgumentException</code> if any of the following are true:
152     * <ul><li>the values array is null</li>
153     *     <li>the weights array is null</li>
154     *     <li>the weights array does not have the same length as the values array</li>
155     *     <li>the weights array contains one or more infinite values</li>
156     *     <li>the weights array contains one or more NaN values</li>
157     *     <li>the weights array contains negative values</li>
158     *     <li>the start and length arguments do not determine a valid array</li>
159     * </ul>
160     *
161     * @param values the input array
162     * @param weights the weights array
163     * @param begin index of the first array element to include
164     * @param length the number of elements to include
165     * @return the mean of the values or Double.NaN if length = 0
166     * @throws IllegalArgumentException if the parameters are not valid
167     */
168    public double evaluate(final double[] values, final double[] weights,
169                           final int begin, final int length) {
170        if (test(values, weights, begin, length)) {
171            Sum sum = new Sum();
172
173            // Compute initial estimate using definitional formula
174            double sumw = sum.evaluate(weights,begin,length);
175            double xbarw = sum.evaluate(values, weights, begin, length) / sumw;
176
177            // Compute correction factor in second pass
178            double correction = 0;
179            for (int i = begin; i < begin + length; i++) {
180                correction += weights[i] * (values[i] - xbarw);
181            }
182            return xbarw + (correction/sumw);
183        }
184        return Double.NaN;
185    }
186
187    @Override
188    public void readFrom(StreamInput in) throws IOException {
189        moment = new FirstMoment();
190        moment.readFrom(in);
191        incMoment = in.readBoolean();
192    }
193
194    @Override
195    public void writeTo(StreamOutput out) throws IOException {
196        moment.writeTo(out);
197        out.writeBoolean(incMoment);
198    }
199}
