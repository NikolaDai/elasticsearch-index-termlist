024/**
025 * Computes the variance of the available values.  By default, the unbiased
026 * "sample variance" definitional formula is used:
027 *
028 * variance = sum((x_i - mean)^2) / (n - 1)
029 *
030 * where mean is the {@link Mean} and <code>n</code> is the number
031 * of sample observations.
032 *
033 * The definitional formula does not have good numerical properties, so
034 * this implementation does not compute the statistic using the definitional
035 * formula.
036 * <ul><li> The <code>getResult</code> method computes the variance using
037 * updating formulas based on West's algorithm, as described in
038 * <a href="http://doi.acm.org/10.1145/359146.359152"> Chan, T. F. and
039 * J. G. Lewis 1979, <i>Communications of the ACM</i>,
040 * vol. 22 no. 9, pp. 526-531.</a></li>
041 * <li> The <code>evaluate</code> methods leverage the fact that they have the
042 * full array of values in memory to execute a two-pass algorithm.
043 * Specifically, these methods use the "corrected two-pass algorithm" from
044 * Chan, Golub, Levesque, <i>Algorithms for Computing the Sample Variance</i>,
045 * American Statistician, vol. 37, no. 3 (1983) pp. 242-247.</li></ul>
046 * Note that adding values using <code>increment</code> or
047 * <code>incrementAll</code> and then executing <code>getResult</code> will
048 * sometimes give a different, less accurate, result than executing
049 * <code>evaluate</code> with the full array of values. The former approach
050 * should only be used when the full array of values is not available.
051 *
052 * The "population variance"  ( sum((x_i - mean)^2) / n ) can also
053 * be computed using this statistic.  The <code>isBiasCorrected</code>
054 * property determines whether the "population" or "sample" value is
055 * returned by the <code>evaluate</code> and <code>getResult</code> methods.
056 * To compute population variances, set this property to <code>false.</code>
057 *
058 *
059 * <strong>Note that this implementation is not synchronized.</strong> If
060 * multiple threads access an instance of this class concurrently, and at least
061 * one of the threads invokes the <code>increment()</code> or
062 * <code>clear()</code> method, it must be synchronized externally.
063 */
064public class Variance extends AbstractStorelessUnivariateStatistic {
065
066    /** SecondMoment is used in incremental calculation of Variance*/
067    protected SecondMoment moment = null;
068
069    /**
070     * Whether or not {@link #increment(double)} should increment
071     * the internal second moment. When a Variance is constructed with an
072     * external SecondMoment as a constructor parameter, this property is
073     * set to false and increments must be applied to the second moment
074     * directly.
075     */
076    protected boolean incMoment = true;
077
078    /**
079     * Whether or not bias correction is applied when computing the
080     * value of the statistic. True means that bias is corrected.  See
081     * {@link Variance} for details on the formula.
082     */
083    private boolean isBiasCorrected = true;
084
085    /**
086     * Constructs a Variance based on an external second moment.
087     * When this constructor is used, the statistic may only be
088     * incremented via the moment, i.e., {@link #increment(double)}
089     * does nothing; whereas {@code m2.increment(value)} increments
090     * both {@code m2} and the Variance instance constructed from it.
091     *
092     * @param m2 the SecondMoment (Third or Fourth moments work
093     * here as well.)
094     */
095    public Variance(final SecondMoment m2) {
096        incMoment = false;
097        this.moment = m2;
098    }
099
100    /**
101     * Constructs a Variance with the specified <code>isBiasCorrected</code>
102     * property
103     *
104     * @param isBiasCorrected  setting for bias correction - true means
105     * bias will be corrected and is equivalent to using the argumentless
106     * constructor
107     */
108    public Variance(boolean isBiasCorrected) {
109        moment = new SecondMoment();
110        this.isBiasCorrected = isBiasCorrected;
111    }
112
113    /**
114     * If all values are available, it is more accurate to use
115     * {@link #evaluate(double[])} rather than adding values one at a time
116     * using this method and then executing {@link #getResult}, since
117     * <code>evaluate</code> leverages the fact that is has the full
118     * list of values together to execute a two-pass algorithm.
119     * See {@link Variance}.
120     *
121     * Note also that when {@link #Variance(SecondMoment)} is used to
122     * create a Variance, this method does nothing. In that case, the
123     * SecondMoment should be incremented directly.
124     */
125    @Override
126    public void increment(final double d) {
127        if (incMoment) {
128            moment.increment(d);
129        }
130    }
131
132    @Override
133    public double getResult() {
134        if (moment.n == 0) {
135            return Double.NaN;
136        } else if (moment.n == 1) {
137            return 0d;
138        } else {
139            if (isBiasCorrected) {
140                return moment.m2 / (moment.n - 1d);
141            } else {
142                return moment.m2 / (moment.n);
143            }
144        }
145    }
146
147    public long getN() {
148        return moment.getN();
149    }
150
151    @Override
152    public void clear() {
153        if (incMoment) {
154            moment.clear();
155        }
156    }

158    /**
159     * Returns the variance of the entries in the input array, or
160     * <code>Double.NaN</code> if the array is empty.
161     *
162     * See {@link Variance} for details on the computing algorithm.
163     *
164     * Returns 0 for a single-value (i.e. length = 1) sample.
165     *
166     * Throws <code>IllegalArgumentException</code> if the array is null.
167     *
168     * Does not change the internal state of the statistic.
169     *
170     * @param values the input array
171     * @return the variance of the values or Double.NaN if length = 0
172     * @throws IllegalArgumentException if the array is null
173     */
174    @Override
175    public double evaluate(final double[] values) {
176        if (values == null) {
177            throw new IllegalArgumentException("values == null");
178        }
179        return evaluate(values, 0, values.length);
180    }

182    /**
183     * Returns the variance of the entries in the specified portion of
184     * the input array, or <code>Double.NaN</code> if the designated subarray
185     * is empty.
186     * <p>
187     * See {@link Variance} for details on the computing algorithm.</p>
188     * <p>
189     * Returns 0 for a single-value (i.e. length = 1) sample.</p>
190     * <p>
191     * Does not change the internal state of the statistic.</p>
192     * <p>
193     * Throws <code>IllegalArgumentException</code> if the array is null.</p>
194     *
195     * @param values the input array
196     * @param begin index of the first array element to include
197     * @param length the number of elements to include
198     * @return the variance of the values or Double.NaN if length = 0
199     * @throws IllegalArgumentException if the array is null or the array index
200     *  parameters are not valid
201     */
202    @Override
203    public double evaluate(final double[] values, final int begin, final int length) {
204        double var = Double.NaN;
205        if (test(values, begin, length)) {
206            clear();
207            if (length == 1) {
208                var = 0.0;
209            } else if (length > 1) {
210                Mean mean = new Mean();
211                double m = mean.evaluate(values, begin, length);
212                var = evaluate(values, m, begin, length);
213            }
214        }
215        return var;
216    }
217
218    /**
219     * Returns the weighted variance of the entries in the specified portion of
220     * the input array, or <code>Double.NaN</code> if the designated subarray
221     * is empty.
222     *
223     * Uses the formula
224     * <pre>
225     *   Sigma(weights[i]*(values[i] - weightedMean)^2)/(Sigma(weights[i]) - 1)
226     * </pre>
227     * where weightedMean is the weighted mean
228     *
229     * This formula will not return the same result as the unweighted variance when all
230     * weights are equal, unless all weights are equal to 1. The formula assumes that
231     * weights are to be treated as "expansion values," as will be the case if for example
232     * the weights represent frequency counts. To normalize weights so that the denominator
233     * in the variance computation equals the length of the input vector minus one, use <pre>
234     *   <code>evaluate(values, MathArrays.normalizeArray(weights, values.length)); </code>
235     * </pre>
236     *
237     * Returns 0 for a single-value (i.e. length = 1) sample.
238     *
239     * Throws <code>IllegalArgumentException</code> if any of the following are true:
240     * <ul><li>the values array is null</li>
241     *     <li>the weights array is null</li>
242     *     <li>the weights array does not have the same length as the values array</li>
243     *     <li>the weights array contains one or more infinite values</li>
244     *     <li>the weights array contains one or more NaN values</li>
245     *     <li>the weights array contains negative values</li>
246     *     <li>the start and length arguments do not determine a valid array</li>
247     * </ul>
248     *
249     * Does not change the internal state of the statistic.
250     *
251     * Throws <code>IllegalArgumentException</code> if either array is null.
252     *
253     * @param values the input array
254     * @param weights the weights array
255     * @param begin index of the first array element to include
256     * @param length the number of elements to include
257     * @return the weighted variance of the values or Double.NaN if length = 0
258     * @throws IllegalArgumentException if the parameters are not valid
259     */
260    public double evaluate(final double[] values, final double[] weights,
261                           final int begin, final int length) {
262        double var = Double.NaN;
263        if (test(values, weights,begin, length)) {
264            clear();
265            if (length == 1) {
266                var = 0.0;
267            } else if (length > 1) {
268                Mean mean = new Mean();
269                double m = mean.evaluate(values, weights, begin, length);
270                var = evaluate(values, weights, m, begin, length);
271            }
272        }
273        return var;
274    }

278     * Returns the weighted variance of the entries in the the input array.
279     *
280     * Uses the formula
281     * <pre>
282     *   Sigma(weights[i]*(values[i] - weightedMean)^2)/(Sigma(weights[i]) - 1)
283     * </pre>
284     * where weightedMean is the weighted mean
285     *
286     * This formula will not return the same result as the unweighted variance when all

288     * weights are to be treated as "expansion values," as will be the case if for example
289     * the weights represent frequency counts. To normalize weights so that the denominator
290     * in the variance computation equals the length of the input vector minus one, use <pre>
291     *   <code>evaluate(values, MathArrays.normalizeArray(weights, values.length)); </code>
292     * </pre>
293     *
294     * Returns 0 for a single-value (i.e. length = 1) sample.
295     *
296     * Throws <code>IllegalArgumentException</code> if any of the following are true:
297     * <ul><li>the values array is null</li>
298     *     <li>the weights array is null</li>
299     *     <li>the weights array does not have the same length as the values array</li>
300     *     <li>the weights array contains one or more infinite values</li>
301     *     <li>the weights array contains one or more NaN values</li>
302     *     <li>the weights array contains negative values</li>
303     * </ul>
304     *
305     * Does not change the internal state of the statistic.
306     *
307     * Throws <code>IllegalArgumentException</code> if either array is null.

315        return evaluate(values, weights, 0, values.length);

318    /**
319     * Returns the variance of the entries in the specified portion of
320     * the input array, using the precomputed mean value.  Returns
321     * <code>Double.NaN</code> if the designated subarray is empty.
322     *
323     * See {@link Variance} for details on the computing algorithm.
324     *
325     * The formula used assumes that the supplied mean value is the arithmetic
326     * mean of the sample data, not a known population parameter.  This method
327     * is supplied only to save computation when the mean has already been
328     * computed.
329     *
330     * Returns 0 for a single-value (i.e. length = 1) sample.
331     *
332     * Throws <code>IllegalArgumentException</code> if the array is null.
333     *
334     * Does not change the internal state of the statistic.
335     *
336     * @param values the input array
337     * @param mean the precomputed mean value
338     * @param begin index of the first array element to include
339     * @param length the number of elements to include
340     * @return the variance of the values or Double.NaN if length = 0
341     * @throws IllegalArgumentException if the array is null or the array index
342     *  parameters are not valid
343     */
344    public double evaluate(final double[] values, final double mean,
345            final int begin, final int length) {
346        double var = Double.NaN;
347        if (test(values, begin, length)) {
348            if (length == 1) {
349                var = 0.0;
350            } else if (length > 1) {
351                double accum = 0.0;
352                double dev = 0.0;
353                double accum2 = 0.0;
354                for (int i = begin; i < begin + length; i++) {
355                    dev = values[i] - mean;
356                    accum += dev * dev;
357                    accum2 += dev;
358                }
359                if (isBiasCorrected) {
360                    var = (accum - (accum2 * accum2 / (double) length)) / ((double) length - 1.0);
361                } else {
362                    var = (accum - (accum2 * accum2 / (double) length)) / (double) length;
363                }
364            }
365        }
366        return var;
367    }

370     * Returns the variance of the entries in the input array, using the
371     * precomputed mean value.  Returns <code>Double.NaN</code> if the array
372     * is empty.
373     *
374     * See {@link Variance} for details on the computing algorithm.
375     *
376     * If <code>isBiasCorrected</code> is <code>true</code> the formula used
377     * assumes that the supplied mean value is the arithmetic mean of the
378     * sample data, not a known population parameter.  If the mean is a known
379     * population parameter, or if the "population" version of the variance is
380     * desired, set <code>isBiasCorrected</code> to <code>false</code> before
381     * invoking this method.
382     *
383     * Returns 0 for a single-value (i.e. length = 1) sample.
384     * Throws <code>IllegalArgumentException</code> if the array is null.
385     * Does not change the internal state of the statistic.
386     *
387     * @param values the input array
388     * @param mean the precomputed mean value
389     * @return the variance of the values or Double.NaN if the array is empty
390     * @throws IllegalArgumentException if the array is null
391     */
392    public double evaluate(final double[] values, final double mean) {
393        return evaluate(values, mean, 0, values.length);
394    }

396    /**
397     * Returns the weighted variance of the entries in the specified portion of
398     * the input array, using the precomputed weighted mean value.  Returns
399     * <code>Double.NaN</code> if the designated subarray is empty.
400     *
401     * Uses the formula <pre>
402     *   Sigma(weights[i]*(values[i] - mean)^2)/(Sigma(weights[i]) - 1)
403     * </pre>
404     *
405     * The formula used assumes that the supplied mean value is the weighted arithmetic
406     * mean of the sample data, not a known population parameter. This method
407     * is supplied only to save computation when the mean has already been
408     * computed.
409     *
410     * This formula will not return the same result as the unweighted variance when all
411     * weights are equal, unless all weights are equal to 1. The formula assumes that
412     * weights are to be treated as "expansion values," as will be the case if for example
413     * the weights represent frequency counts. To normalize weights so that the denominator
414     * in the variance computation equals the length of the input vector minus one, use <pre>
415     *   <code>evaluate(values, MathArrays.normalizeArray(weights, values.length), mean); </code>
416     * </pre>
417     *
418     * Returns 0 for a single-value (i.e. length = 1) sample.
419     *
420     * Throws <code>IllegalArgumentException</code> if any of the following are true:
421     * <ul><li>the values array is null</li>
422     *     <li>the weights array is null</li>
423     *     <li>the weights array does not have the same length as the values array</li>
424     *     <li>the weights array contains one or more infinite values</li>
425     *     <li>the weights array contains one or more NaN values</li>
426     *     <li>the weights array contains negative values</li>
427     *     <li>the start and length arguments do not determine a valid array</li>
428     * </ul>
429     *
430     * Does not change the internal state of the statistic.
431     *
432     * @param values the input array
433     * @param weights the weights array
434     * @param mean the precomputed weighted mean value
435     * @param begin index of the first array element to include
436     * @param length the number of elements to include
437     * @return the variance of the values or Double.NaN if length = 0
438     * @throws IllegalArgumentException if the parameters are not valid
439     */
440    public double evaluate(final double[] values, final double[] weights,
441                           final double mean, final int begin, final int length) {
442        double var = Double.NaN;
443        if (test(values, weights, begin, length)) {
444            if (length == 1) {
445                var = 0.0;
446            } else if (length > 1) {
447                double accum = 0.0;
448                double dev = 0.0;
449                double accum2 = 0.0;
450                for (int i = begin; i < begin + length; i++) {
451                    dev = values[i] - mean;
452                    accum += weights[i] * (dev * dev);
453                    accum2 += weights[i] * dev;
454                }
455                double sumWts = 0;
456                for (int i = begin; i < begin + length; i++) {
457                    sumWts += weights[i];
458                }
459                if (isBiasCorrected) {
460                    var = (accum - (accum2 * accum2 / sumWts)) / (sumWts - 1.0);
461                } else {
462                    var = (accum - (accum2 * accum2 / sumWts)) / sumWts;
463                }
464            }
465        }
466        return var;
467    }

527    @Override
528    public void readFrom(StreamInput in) throws IOException {
529        moment = new SecondMoment();
530        moment.readFrom(in);
531        incMoment = in.readBoolean();
532        isBiasCorrected = in.readBoolean();
533    }

535    @Override
536    public void writeTo(StreamOutput out) throws IOException {
537        moment.writeTo(out);
538        out.writeBoolean(incMoment);
