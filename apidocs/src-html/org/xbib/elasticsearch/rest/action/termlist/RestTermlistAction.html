001package org.xbib.elasticsearch.rest.action.termlist;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.Strings;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.common.xcontent.ToXContent;
008import org.elasticsearch.common.xcontent.XContentBuilder;
009import org.elasticsearch.rest.BaseRestHandler;
010import org.elasticsearch.rest.BytesRestResponse;
011import org.elasticsearch.rest.RestChannel;
012import org.elasticsearch.rest.RestController;
013import org.elasticsearch.rest.RestRequest;
014import org.elasticsearch.rest.RestResponse;
015import org.elasticsearch.rest.RestStatus;
016import org.elasticsearch.rest.action.support.RestBuilderListener;
017import org.xbib.elasticsearch.action.termlist.TermInfo;
018import org.xbib.elasticsearch.action.termlist.TermlistAction;
019import org.xbib.elasticsearch.action.termlist.TermlistRequest;
020import org.xbib.elasticsearch.action.termlist.TermlistResponse;
021
022import java.util.Map;
023
024import static org.elasticsearch.rest.RestRequest.Method.GET;
025import static org.elasticsearch.rest.action.support.RestActions.buildBroadcastShardsHeader;
026
027public class RestTermlistAction extends BaseRestHandler {
028
029    @Inject
030    public RestTermlistAction(Settings settings, Client client, RestController controller) {
031        super(settings, controller, client);
032        controller.registerHandler(GET, "/_termlist", this);
033        controller.registerHandler(GET, "/{index}/_termlist", this);
034    }
035
036    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
037        try {
038            TermlistRequest termlistRequest = new TermlistRequest(Strings.splitStringByCommaToArray(request.param("index")));
039            termlistRequest.setField(request.param("field"));
040            termlistRequest.setTerm(request.param("term"));
041            termlistRequest.setFrom(request.paramAsInt("from", 0));
042            termlistRequest.setSize(request.paramAsInt("size", -1));
043            termlistRequest.sortByDocFreq(request.paramAsBoolean("sortbydocfreqs", false));
044            termlistRequest.sortByTotalFreq(request.paramAsBoolean("sortbytotalfreqs", false));
045            termlistRequest.sortByTerm(request.paramAsBoolean("sortbyterms", false));
046            final long t0 = System.nanoTime();
047            client.execute(TermlistAction.INSTANCE, termlistRequest, new RestBuilderListener<TermlistResponse>(channel) {
048                @Override
049                public RestResponse buildResponse(TermlistResponse response, XContentBuilder builder) throws Exception {
050                    builder.startObject();
051                    buildBroadcastShardsHeader(builder, response);
052                    builder.field("took", (System.nanoTime() - t0) / 1000000);
053                    builder.field("numdocs", response.getNumDocs());
054                    builder.field("numterms", response.getTermlist().size());
055                    builder.startArray("terms");
056                    for (Map.Entry<String, TermInfo> t : response.getTermlist().entrySet()) {
057                        builder.startObject().field("term", t.getKey());
058                        t.getValue().toXContent(builder, ToXContent.EMPTY_PARAMS);
059                        builder.endObject();
060                    }
061                    builder.endArray();
062                    builder.endObject();
063                    return new BytesRestResponse(RestStatus.OK, builder);
064                }
065            });
066        } catch (Throwable t) {
067            logger.error(t.getMessage(), t);
068            channel.sendResponse(new BytesRestResponse(RestStatus.INTERNAL_SERVER_ERROR, t.getMessage()));
069        }
070    }
071}
